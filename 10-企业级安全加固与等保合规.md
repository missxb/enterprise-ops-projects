### 1.1 软件供应链安全（SBOM / SLSA / Sigstore）

> 软件供应链安全是等保3.0和关键信息基础设施安全保护的重要组成部分。通过构建可验证的软件构建与分发链路，确保从源码到部署的全链路可信。

#### SBOM（软件物料清单）

```bash
# 使用 syft 生成镜像 SBOM（SPDX 格式）
syft myregistry.io/myapp:v1.2.3 -o spdx-json > sbom-spdx.json

# 使用 syft 生成目录 SBOM（CycloneDX 格式）
syft ./src -o cyclonedx-json > sbom-cdx.json

# 使用 grype 扫描 SBOM 中的已知漏洞
grype sbom:./sbom-spdx.json --fail-on high
```

#### SLSA（供应链完整性框架）

```yaml
# GitHub Actions: SLSA Level 3 构建示例（使用 slsa-github-generator）
# .github/workflows/release.yml
name: SLSA Build
on:
  push:
    tags: ['v*']
jobs:
  build:
    runs-on: ubuntu-latest
    outputs:
      hashes: ${{ steps.hash.outputs.hashes }}
    steps:
      - uses: actions/checkout@v4
      - name: Build binary
        run: make build
      - name: Generate checksums
        id: hash
        run: |
          sha256sum dist/* > checksums.txt
          echo "hashes=$(sha256sum dist/* | base64 -w0)" >> "$GITHUB_OUTPUT"
      - name: Upload artifact
        uses: actions/upload-artifact@v4
        with:
          name: release
          path: dist/

  provenance:
    needs: build
    permissions:
      actions: read
      id-token: write
      contents: write
    uses: slsa-framework/slsa-github-generator/.github/workflows/generator_generic_slsa3.yml@v2.1.0
    with:
      base-name: "myorg/myapp"
      upload-assets: true
```

#### 应用程序集成 mTLS

```go
// Go 应用使用 SPIFFE Workload API 获取 SVID 并建立 mTLS
package main

import (
    "crypto/tls"
    "fmt"
    "net/http"

    "github.com/spiffe/go-spiffe/v2/spiffetls/tlsconfig"
    "github.com/spiffe/go-spiffe/v2/workloadapi"
)

func main() {
    // 创建 Workload API 客户端，获取 SVID
    ctx := context.Background()
    source, err := workloadapi.NewX509Source(ctx,
        workloadapi.WithClientOptions(
            workloadapi.WithAddr("unix:///run/spire/sockets/agent.sock"),
        ),
    )
    if err != nil {
        panic(err)
    }
    defer source.Close()

    // 使用 SVID 建立 mTLS 服务器
    tlsConfig := tlsconfig.TLSServerConfig(source)
    tlsConfig.ClientAuth = tls.RequireAndVerifyClientCert // 双向验证
    tlsConfig.ClientCAs = tlsconfig.AuthorizeSPIFFE("spiffe://example.com")

    server := &http.Server{
        Addr:      ":8443",
        TLSConfig: tlsConfig,
    }
    fmt.Println("mTLS server listening on :8443")
    server.ListenAndServeTLS("", "")
}
```

#### 安全基线检查脚本

```bash
#!/bin/bash
# cloud-security-check.sh — 云计算安全基线检查（等保3.0 云扩展）

set -euo pipefail

echo "========== 1. 身份认证与访问控制 =========="
# 检查 IAM 策略是否启用 MFA
aws iam generate-credential-report 2>/dev/null || true
sleep 2
aws iam get-credential-report --query 'Content' --output text | base64 -d | \
  awk -F',' 'NR==1 || $4=="true" {print $1, "MFA:", $4}'

# 检查 root 账户是否启用 MFA
ROOT_MFA=$(aws iam get-credential-report --query 'Content' --output text | \
  base64 -d | grep ':root:' | awk -F',' '{print $4}')
echo "Root 账户 MFA: $ROOT_MFA"
if [ "$ROOT_MFA" = "false" ]; then
  echo "[风险] Root 账户未启用 MFA，违反等保要求"
fi

echo "========== 2. 网络安全 =========="
# 检查安全组是否允许 0.0.0.0/0 访问敏感端口
for port in 22 3306 6379 27017 9200; do
  OPEN_SG=$(aws ec2 describe-security-groups \
    --filters Name=ip-permission.cidr,Values="0.0.0.0/0" \
              Name=ip-permission.from-port,Values="$port" \
    --query 'SecurityGroups[].GroupId' --output text 2>/dev/null || true)
  if [ -n "$OPEN_SG" ]; then
    echo "[风险] 端口 $port 对公网开放: $OPEN_SG"
  fi
done

# 检查 VPC Flow Logs 是否启用
VPC_IDS=$(aws ec2 describe-vpcs --query 'Vpcs[].VpcId' --output text)
for vpc in $VPC_IDS; do
  FLOW_LOG=$(aws ec2 describe-flow-logs \
    --filter Name=resource-id,Values="$vpc" \
    --query 'FlowLogs[].FlowLogId' --output text 2>/dev/null || true)
  if [ -z "$FLOW_LOG" ]; then
    echo "[风险] VPC $vpc 未启用 Flow Logs"
  fi
done

echo "========== 3. 存储安全 =========="
# 检查 S3 存储桶是否开启加密
for bucket in $(aws s3api list-buckets --query 'Buckets[].Name' --output text); do
  ENCRYPTION=$(aws s3api get-bucket-encryption --bucket "$bucket" \
    --query 'ServerSideEncryptionConfiguration.Rules[0].ApplyServerSideEncryptionByDefault.SSEAlgorithm' \
    --output text 2>/dev/null || echo "NONE")
  if [ "$ENCRYPTION" = "NONE" ]; then
    echo "[风险] 存储桶 $bucket 未启用加密"
  fi
done

echo "========== 4. 日志与审计 =========="
# 检查 CloudTrail 是否启用多区域跟踪
TRAIL_STATUS=$(aws cloudtrail describe-trails --query 'trailList[].IsMultiRegionTrail' --output text 2>/dev/null || echo "")
if echo "$TRAIL_STATUS" | grep -q "False"; then
  echo "[风险] CloudTrail 未启用多区域跟踪"
fi

echo "========== 5. 密钥管理 =========="
# 检查 KMS 密钥自动轮换
for key_id in $(aws kms list-keys --query 'Keys[].KeyId' --output text); do
  ROTATION=$(aws kms get-key-rotation-status --key-id "$key_id" \
    --query 'KeyRotationEnabled' --output text 2>/dev/null || echo "False")
  if [ "$ROTATION" = "False" ]; then
    echo "[信息] KMS 密钥 $key_id 未启用自动轮换"
  fi
done

echo "========== 检查完成 =========="
```

## 二、主机安全加固

### 2.1 Linux基线加固

```bash
#!/bin/bash
# hardening.sh - CentOS 7/8 主机安全加固

set -euo pipefail

echo "========== 1. 账户安全 =========="
# 禁用root远程登录（保留sudo）
# [修复] 使用更通用的正则匹配，覆盖更多sshd_config变体(注释/未注释/空格差异)
# ^#?PermitRootLogin → 匹配 PermitRootLogin 和 #PermitRootLogin
# ^#?\s*PermitRootLogin → 额外允许空格变体(如 "  PermitRootLogin")
sed -i 's/^#\?PermitRootLogin.*/PermitRootLogin prohibit-password/' /etc/ssh/sshd_config
sed -i 's/^#\?PasswordAuthentication.*/PasswordAuthentication no/' /etc/ssh/sshd_config
sed -i 's/^#\?MaxAuthTries.*/MaxAuthTries 3/' /etc/ssh/sshd_config
sed -i 's/^#\?ClientAliveInterval.*/ClientAliveInterval 300/' /etc/ssh/sshd_config
sed -i 's/^#\?ClientAliveCountMax.*/ClientAliveCountMax 2/' /etc/ssh/sshd_config
# [已修复] 添加SSH登录Banner(等保三级要求)
echo " Authorized access only. All activities are monitored." > /etc/ssh/banner
sed -i 's/^#\?Banner.*/Banner \/etc\/ssh\/banner/' /etc/ssh/sshd_config
systemctl restart sshd

# 密码策略
# [注意] login.defs设置仅影响新创建/修改的密码，已有密码不受影响
# 需要强制已有用户修改密码: chage -d 0 <username>
cat >> /etc/login.defs << 'EOF'
PASS_MAX_DAYS   90
PASS_MIN_DAYS   7
PASS_MIN_LEN    12
PASS_WARN_AGE   14
EOF

# 锁定无用账户
for user in games ftp; do
  passwd -l $user 2>/dev/null || true
done

echo "========== 2. 文件权限加固 =========="
chmod 600 /etc/shadow
chmod 644 /etc/passwd
chmod 600 /etc/ssh/sshd_config
chmod 700 /root
chmod 600 /boot/grub2/grub.cfg

# 查找并修复SUID/SGID文件
find / -type f \( -perm -4000 -o -perm -2000 \) -exec ls -la {} \; 2>/dev/null

echo "========== 3. 内核安全加固 =========="
cat >> /etc/sysctl.d/99-security.conf << 'EOF'
# IP转发（K8s节点必须开启，非K8s节点可设为0）
net.ipv4.ip_forward = 1

# 禁止ICMP重定向
net.ipv4.conf.all.accept_redirects = 0
net.ipv4.conf.all.send_redirects = 0

# 禁止源路由
net.ipv4.conf.all.accept_source_route = 0

# SYN Flood防护
net.ipv4.tcp_syncookies = 1
net.ipv4.tcp_max_syn_backlog = 65535
net.ipv4.tcp_synack_retries = 2

# 禁止ICMP广播
net.ipv4.icmp_echo_ignore_broadcasts = 1

# 记录可疑数据包
net.ipv4.conf.all.log_martians = 1

# 反向路径过滤
net.ipv4.conf.all.rp_filter = 1

# 禁止IP伪造
net.ipv6.conf.all.accept_redirects = 0
net.ipv4.conf.all.accept_source_route = 0
net.ipv6.conf.all.accept_source_route = 0
EOF
sysctl --system

# [安全加固] SELinux强制模式(非K8s节点)
# 注意: K8s节点不建议开启SELinux enforcing，因为K8s的CNI插件、
# 容器运行时等可能与SELinux策略冲突，导致Pod无法正常启动。
# 非K8s节点应强制启用SELinux enforcing以满足等保三级要求。
if ! systemctl is-active kubelet &>/dev/null; then
  # 设置SELinux为enforcing模式(永久生效)
  sed -i 's/^SELINUX=.*/SELINUX=enforcing/' /etc/selinux/config
  # 当前会话立即生效(不需要重启)
  setenforce 1
  echo "[INFO] SELinux已设置为enforcing模式(非K8s节点)"
else
  echo "[SKIP] K8s节点跳过SELinux enforcing设置(与K8s CNI/运行时可能冲突)"
  echo "[NOTE] K8s节点可考虑使用SELinux permissive模式进行审计日志记录"
fi

echo "========== 4. 防火墙配置 =========="
# 使用firewalld而非iptables
# [注意] 如果节点运行K8s，不应启用firewalld，K8s的kube-proxy和CNI需要直接操作iptables
# systemctl enable firewalld
# systemctl start firewalld

# [K8s节点跳过] 如果节点运行K8s，不应执行firewall-cmd
# [K8s替代方案] K8s节点应使用NetworkPolicy替代firewalld作为防火墙
# NetworkPolicy提供Pod级别的网络隔离，是K8s原生的网络防火墙机制
# 示例: 拒绝所有入站流量
#   apiVersion: networking.k8s.io/v1
#   kind: NetworkPolicy
#   spec:
#     podSelector: {}
#     policyTypes: ["Ingress", "Egress"]
# 注意: NetworkPolicy需要CNI插件支持(Calico/Cilium/Weave等)
# Flannel不支持NetworkPolicy，需更换为Calico或Cilium
if ! systemctl is-active kubelet &>/dev/null; then
# 默认拒绝所有
firewall-cmd --set-default-zone=drop

# 允许SSH
firewall-cmd --permanent --add-service=ssh

# 允许HTTP/HTTPS
firewall-cmd --permanent --add-port=80/tcp
firewall-cmd --permanent --add-port=443/tcp

# 允许内网管理
firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="10.10.0.0/16" accept'

# [修复] 添加数据库端口限制(仅允许内网访问)
firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="10.10.0.0/16" port port="3306" protocol="tcp" accept'
firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="10.10.0.0/16" port port="6379" protocol="tcp" accept'
# 拒绝外部访问数据库端口
```

#### 2.3.1 Pod安全标准（Pod Security Standards）

Pod Security Admission是K8s 1.25+原生方案，替代已废弃的PodSecurityPolicy。三个安全级别：
- **privileged**：不受限制，适用于系统组件
- **baseline**：最小限制，禁止已知特权提升
- **restricted**：严格限制，适用于生产工作负载

```yaml

    pod-security.kubernetes.io/audit: restricted
    pod-security.kubernetes.io/warn: restricted

  template:
```

#### 2.3.4 OPA/Gatekeeper策略即代码

> Pod Security Admission是K8s原生方案，OPA/Gatekeeper提供更灵活的策略控制。

```bash
#!/bin/bash
# data_encryption.sh - 数据加密方案

echo "========== 磁盘加密 (LUKS) =========="
# [修复] cryptsetup luksFormat默认交互式确认,脚本中应使用--batch-mode避免挂起
# 创建加密分区(非交互式)
cryptsetup luksFormat --batch-mode --type luks2 /dev/sdb1
cryptsetup luksOpen --batch-mode /dev/sdb1 encrypted_data
mkfs.ext4 /dev/mapper/encrypted_data
mkdir -p /data/encrypted
mount /dev/mapper/encrypted_data /data/encrypted

# 添加恢复密钥（等保要求:密钥备份与恢复能力）
# 使用HashiCorp Vault集中管理LUKS密钥
vault kv put secret/luks-keys/server-01 \
    key="$(dd if=/dev/urandom bs=32 count=1 2>/dev/null | base64)" \
    slot=0

# 非交互式自动解锁脚本（使用Key File方式）
cat > /etc/cryptsetup-keys.d/encrypted_data.key << 'KEYEOF'
$(dd if=/dev/urandom bs=64 count=1 2>/dev/null | base64)
KEYEOF
chmod 000 /etc/cryptsetup-keys.d/encrypted_data.key
cryptsetup luksAddKey /dev/sdb1 /etc/cryptsetup-keys.d/encrypted_data.key --batch-mode
# 配置系统启动自动挂载(使用密钥文件非交互解锁):

echo "========== TLS证书管理 =========="
```

#### 2.3.5 镜像签名验证（Sigstore/Cosign）

```bash
# 安装Cosign
curl -L -o /usr/local/bin/cosign https://github.com/sigstore/cosign/releases/latest/download/cosign-linux-amd64
chmod +x /usr/local/bin/cosign

# 签名镜像
cosign sign --key cosign.key registry.example.com/app:v1.0

# 验证签名
cosign verify --key cosign.pub registry.example.com/app:v1.0
```

> Cosign是Sigstore项目的镜像签名工具，支持密钥签名和无密钥签名(Fulcio+Rekor)。生产环境建议在CI/CD中自动签名，K8s中通过Kyverno/OPA策略强制验证签名。

#### Kyverno镜像签名验证(强制策略)

> Kyverno是K8s原生的策略引擎,使用YAML而非OPA/Rego编写策略,学习曲线更低。

```yaml
# kyverno-image-verify.yaml
# 安装Kyverno:
# helm install kyverno kyverno/kyverno -n kyverno --create-namespace

# 策略1: 强制验证镜像签名(禁止未签名镜像部署到生产环境)
apiVersion: kyverno.io/v1
kind: ClusterPolicy
metadata:
  name: verify-image-signature
  annotations:
    policies.kyverno.io/title: 镜像签名验证
    policies.kyverno.io/description: 所有镜像必须经过Cosign签名验证
spec:
  validationFailureAction: Enforce  # Enforce=阻断, Audit=仅告警
  background: false
  rules:
    - name: verify-cosign-signature
      match:
        any:
          - resources:
              kinds:
                - Pod
              namespaces:
                - production
      verifyImages:
        - imageReferences:
            - "harbor.internal.com/*"
          attestors:
            - entries:
                - keys:
                    # Cosign公钥(需替换为实际公钥)
                    publicKeys: |-
                      -----BEGIN PUBLIC KEY-----
                      MFkwEwYHKoZIzj0CAQYIKoZIzj0DAQcDQgAE...
                      -----END PUBLIC KEY-----
          attestations:
            - type: https://slsa.dev/provenance/v0.2
              conditions:
                - all:
                    - key: "{{builder.id}}"
                      operator: Equals
                      value: "https://github.com/actions/runner"
---
# 策略2: 禁止从公共仓库拉取镜像
apiVersion: kyverno.io/v1
kind: ClusterPolicy
metadata:
  name: restrict-image-registries
  annotations:
    policies.kyverno.io/title: 限制镜像来源
    policies.kyverno.io/description: 仅允许从内部Harbor仓库拉取镜像
spec:
  validationFailureAction: Enforce
  background: true
  rules:
    - name: validate-image-registry
      match:
        any:
          - resources:
              kinds:
                - Pod
              namespaces:
                - production
      validate:
        pattern:
          spec:
            containers:
              - image: "harbor.internal.com/*"
---
# 策略3: 镜像漏洞扫描结果验证(集成Trivy)
apiVersion: kyverno.io/v1
kind: ClusterPolicy
metadata:
  name: scan-image-vulnerabilities
  annotations:
    policies.kyverno.io/title: 镜像漏洞扫描
    policies.kyverno.io/description: 使用Trivy扫描镜像漏洞并拦截高危镜像
spec:
  validationFailureAction: Audit  # 建议先Audit观察,确认无误后改为Enforce
  background: true
  rules:
    - name: scan-and-verify
      match:
        any:
          - resources:
              kinds:
                - Pod
              namespaces:
                - production
      verifyImages:
        - imageReferences:
            - "harbor.internal.com/*"
          attestors:
            - entries:
                - attestor:
                    name: Trivy
                    keyless:
                      url: https://fulcio.sigstore.dev
                      issuer: https://token.actions.githubusercontent.com
                      rekor:
                        url: https://rekor.sigstore.dev
```

## 四、等保合规检查清单

### 4.1 法律依据与版本说明

> **等保2.0 (GB/T 22239-2019)**：2019年12月1日实施，适用于当前绝大多数等保三级测评。
> **等保3.0 (GB/T 22239-2025)**：2025年发布，对云计算、大数据、物联网、移动互联、工业控制系统等新技术新应用提出了更高的安全要求。
>
> 本项目依据：GB/T 22239-2019（等保2.0）编写，适用于2025年之前的测评标准。
> 如需依据等保3.0进行测评，请关注GB/T 22239-2025的新增要求，特别是：
> - 云计算安全扩展要求（云平台、虚拟化安全）
> - 大数据安全扩展要求（数据分类分级、隐私保护）
> - 物联网安全扩展要求（感知节点、网关安全）
>
> **重要提示**：等保三级需要由具有资质的第三方测评机构出具正式测评报告。本项目提供的 compliance-check.sh 脚本仅用于日常自查，不能替代正式的等保测评。自查发现的问题应在正式测评前完成整改。

```bash
# fix_log4j.sh - Log4Shell漏洞修复

# 查找所有Log4j版本
find / -name "log4j-core-*.jar" 2>/dev/null

# 修复方案1: 升级到2.17.0+
# 修复方案2: 设置JVM参数

# 修复方案3: 删除JndiLookup类

# ⚠️ 重要提示：
# -Dlog4j2.formatMsgNoLookups=true 仅是临时缓解措施，不能替代版本升级！
# Log4j 2.17+ 已从根本上修复 JNDI 注入漏洞，且 2.15.0 的 formatMsgNoLookups 存在绕过风险。
#
# 推荐修复优先级：
#   1. [最佳] 升级到 Log4j 2.17.1+（彻底修复所有 CVE）
#   2. [临时] 使用 formatMsgNoLookups 参数，但必须尽快安排版本升级
#   3. [临时] 删除 JndiLookup.class，但可能影响功能

# 验证当前 Log4j 版本
find / -name "log4j-core-*.jar" 2>/dev/null | while read jar; do
  VER=$(unzip -p "$jar" META-INF/MANIFEST.MF 2>/dev/null | grep "Bundle-Version" | awk '{print $2}' | tr -d '\r')
```

## 九、应急响应流程

```
安全事件响应流程(SOP):

1. 检测阶段 (0-15分钟)
   ├── 确认事件真实性
   ├── 评估影响范围
   ├── 通知安全团队
   └── 记录事件时间线

2. 遏制阶段 (15-60分钟)
   ├── 隔离受影响系统
   ├── 封禁攻击IP
   ├── 保留证据(日志/快照)
   └── 阻止横向移动

3. 消除阶段 (1-24小时)
   ├── 修复漏洞
   ├── 清除恶意软件
   ├── 重置泄露凭证
   └── 更新安全策略

4. 恢复阶段 (24-72小时)
   ├── 恢复系统服务
   ├── 验证数据完整性
   ├── 监控异常行为
   └── 逐步恢复业务

5. 总结阶段 (72小时后)
   ├── 编写事件报告
   ├── 复盘改进措施
   ├── 更新应急预案
   └── 安全培训
```

### 9.1 可执行应急响应脚本

```bash
#!/bin/bash
# security-incident-response.sh — 自动化安全事件应急响应脚本
# 用法: ./security-incident-response.sh <action> [options]
#   action:  isolate | block-ip | collect-evidence | restore | full-report
# 示例:
#   ./security-incident-response.sh isolate          # 隔离当前主机
#   ./security-incident-response.sh block-ip 1.2.3.4 # 封禁指定IP
#   ./security-incident-response.sh collect-evidence  # 收集取证数据
#   ./security-incident-response.sh restore            # 恢复服务
#   ./security-incident-response.sh full-report        # 生成事件报告

set -euo pipefail

# ============================================================
# 配置区域
# ============================================================
EVIDENCE_DIR="/evidence/$(date +%Y%m%d-%H%M%S)"
SIEM_WEBHOOK="${SIEM_WEBHOOK:-https://siem.internal/api/v1/events}"
SOAR_API="${SOAR_API:-http://thehive:9000}"
SOAR_API_KEY="${SOAR_API_KEY:-}"
ALERT_EMAIL="security-team@company.com"
LOG_FILE="/var/log/incident-response.log"

# ============================================================
# 通用函数
# ============================================================
log() {
    local level="$1"; shift
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] [$level] $*" | tee -a "$LOG_FILE"
}

notify_soar() {
    local title="$1" description="$2" severity="${3:-high}"
    if [[ -n "$SOAR_API_KEY" ]]; then
        curl -s -X POST "${SOAR_API}/api/case" \
            -H "Authorization: Bearer ${SOAR_API_KEY}" \
            -H "Content-Type: application/json" \
            -d "{
                \"title\": \"${title}\",
                \"description\": \"${description}\",
                \"severity\": ${severity},
                \"flag\": true,
                \"tags\": [\"auto-response\", \"$(hostname)\"]
            }" >> "$LOG_FILE" 2>&1 || true
        log INFO "已创建 SOAR 工单: ${title}"
    fi
}

notify_siem() {
    local event_type="$1" details="$2"
    curl -s -X POST "$SIEM_WEBHOOK" \
        -H "Content-Type: application/json" \
        -d "{\"event_type\": \"${event_type}\", \"host\": \"$(hostname)\", \"details\": \"${details}\", \"timestamp\": \"$(date -Iseconds)\"}" \
        >> "$LOG_FILE" 2>&1 || true
}

# ============================================================
# 功能1: 主机隔离（iptables panic 模式 + 网络断开）
# ============================================================
cmd_isolate() {
    log WARN ">>> 开始隔离主机 $(hostname)"
    notify_soar "主机隔离: $(hostname)" "安全事件触发自动隔离" 3

    # 1. iptables panic 模式 — 阻断所有网络流量
    if command -v iptables &>/dev/null; then
        iptables -P INPUT DROP
        iptables -P FORWARD DROP
        iptables -P OUTPUT DROP
        # 仅放行 SSH 管理端口（可选，紧急恢复用）
        iptables -A INPUT -p tcp --dport 22 -s 10.0.0.0/8 -j ACCEPT
        iptables -A OUTPUT -p tcp --sport 22 -d 10.0.0.0/8 -j ACCEPT
        log INFO "iptables 已设置为 DROP ALL 模式"
    fi

    # 2. firewalld panic 模式（如果使用 firewalld）
    if command -v firewall-cmd &>/dev/null; then
        firewall-cmd --panic-on 2>/dev/null || true
        log INFO "firewalld panic 模式已启用"
    fi

    # 3. 断开所有 SSH 会话（保留当前）
    for session in $(who | grep -v "$(tty)" | awk '{print $2}'); do
        pkill -t "$session" 2>/dev/null || true
    done

    log WARN "主机已隔离 — 仅允许管理网段 SSH"
    notify_siem "host_isolation" "主机 $(hostname) 已被隔离"
}

# ============================================================
# 功能2: 封禁指定 IP
# ============================================================
cmd_block_ip() {
    local target_ip="${1:?用法: $0 block-ip <IP>}"

    log WARN ">>> 封禁 IP: ${target_ip}"

    # iptables 封禁
    iptables -A INPUT -s "$target_ip" -j DROP
    iptables -A OUTPUT -d "$target_ip" -j DROP

    # firewalld rich rule（持久化）
    if command -v firewall-cmd &>/dev/null; then
        firewall-cmd --permanent --add-rich-rule="rule family='ipv4' source address='${target_ip}' reject" 2>/dev/null || true
        firewall-cmd --reload 2>/dev/null || true
    fi

    # fail2ban 手动加入
    if command -v fail2ban-client &>/dev/null; then
        fail2ban-client set sshd banip "$target_ip" 2>/dev/null || true
    fi

    log INFO "IP ${target_ip} 已封禁"
    notify_siem "ip_blocked" "IP ${target_ip} 已被封禁"
    notify_soar "IP封禁: ${target_ip}" "自动封禁可疑IP" 2
}

# ============================================================
# 功能3: 取证数据收集
# ============================================================
cmd_collect_evidence() {
    log INFO ">>> 开始收集取证数据 -> ${EVIDENCE_DIR}"
    mkdir -p "${EVIDENCE_DIR}"

    # 系统状态快照
    hostname > "${EVIDENCE_DIR}/hostname.txt"
    date > "${EVIDENCE_DIR}/timestamp.txt"
    uptime > "${EVIDENCE_DIR}/uptime.txt"
    uname -a > "${EVIDENCE_DIR}/uname.txt"

    # 进程快照
    ps auxwwf > "${EVIDENCE_DIR}/ps_aux.txt"
    ps -eo pid,ppid,cmd,etime,%cpu,%mem --sort=-%cpu | head -50 > "${EVIDENCE_DIR}/top_cpu.txt"
    ss -tlnp > "${EVIDENCE_DIR}/listening_ports.txt"
    ss -tnp > "${EVIDENCE_DIR}/established_connections.txt"
    netstat -an > "${EVIDENCE_DIR}/netstat_all.txt" 2>/dev/null || true

    # 用户和权限
    last -50 > "${EVIDENCE_DIR}/last_logins.txt"
    lastb -20 > "${EVIDENCE_DIR}/failed_logins.txt" 2>/dev/null || true
    cat /etc/passwd > "${EVIDENCE_DIR}/passwd.txt"
    cat /etc/shadow > "${EVIDENCE_DIR}/shadow.txt" 2>/dev/null || true
    cat /etc/sudoers > "${EVIDENCE_DIR}/sudoers.txt" 2>/dev/null || true
    w > "${EVIDENCE_DIR}/who_is_online.txt"

    # 日志（最近 10000 行）
    tail -10000 /var/log/secure > "${EVIDENCE_DIR}/secure.log" 2>/dev/null || true
    tail -10000 /var/log/messages > "${EVIDENCE_DIR}/messages.log" 2>/dev/null || true
    tail -10000 /var/log/audit/audit.log > "${EVIDENCE_DIR}/audit.log" 2>/dev/null || true
    journalctl --since "24 hours ago" --no-pager > "${EVIDENCE_DIR}/journal_24h.txt" 2>/dev/null || true

    # 关键文件哈希（完整性检查）
    find /etc -type f -exec sha256sum {} + > "${EVIDENCE_DIR}/etc_sha256.txt" 2>/dev/null || true

    # crontab 和启动项
    for u in $(cut -d: -f1 /etc/passwd); do
        crontab -l -u "$u" 2>/dev/null >> "${EVIDENCE_DIR}/crontabs_all.txt" || true
    done
    systemctl list-unit-files --state=enabled > "${EVIDENCE_DIR}/enabled_services.txt" 2>/dev/null || true

    # Docker 状态（如存在）
    if command -v docker &>/dev/null; then
        docker ps -a > "${EVIDENCE_DIR}/docker_ps.txt" 2>/dev/null || true
        docker inspect $(docker ps -q) > "${EVIDENCE_DIR}/docker_inspect.json" 2>/dev/null || true
    fi

    # 打包证据
    tar czf "${EVIDENCE_DIR}.tar.gz" -C "$(dirname $EVIDENCE_DIR)" "$(basename $EVIDENCE_DIR)"
    chmod 400 "${EVIDENCE_DIR}.tar.gz"

    log INFO "取证数据已收集: ${EVIDENCE_DIR}.tar.gz"
    notify_siem "evidence_collected" "取证数据已保存到 ${EVIDENCE_DIR}.tar.gz"
}

# ============================================================
# 功能4: 恢复服务
# ============================================================
cmd_restore() {
    log INFO ">>> 开始恢复服务"

    # 恢复网络（清除所有 DROP 规则）
    iptables -P INPUT ACCEPT
    iptables -P FORWARD ACCEPT
    iptables -P OUTPUT ACCEPT
    iptables -F

    if command -v firewall-cmd &>/dev/null; then
        firewall-cmd --panic-off 2>/dev/null || true
    fi

    log INFO "网络规则已恢复"

    # 检查关键服务状态
    for svc in sshd firewalld docker nginx mysql; do
        if systemctl is-enabled "$svc" &>/dev/null; then
            systemctl is-active "$svc" &>/dev/null || {
                systemctl start "$svc"
                log INFO "已启动服务: ${svc}"
            }
        fi
    done

    # 验证系统完整性
    if command -v aide &>/dev/null; then
        log INFO "正在运行 AIDE 完整性检查..."
        aide --check >> "$LOG_FILE" 2>&1 || log WARN "AIDE 检测到文件变更，请检查日志"
    fi

    log INFO "恢复流程完成"
    notify_soar "服务恢复: $(hostname)" "安全事件后服务已恢复" 1
}

# ============================================================
# 功能5: 生成事件报告
# ============================================================
cmd_full_report() {
```

### 案例1: TLS证书过期导致全站不可访问

**故障现象**: 凌晨3点监控告警，所有HTTPS服务返回SSL_ERROR_EXPIRED_CERT_KEY_ERR

**根因分析**:
- cert-manager自动续期证书失败，但未触发告警
- Let's Encrypt证书有效期90天，续期窗口为到期前30天
- cert-manager Pod被OOM Killed，无法执行续期任务

**OOM根因详解**:
- cert-manager默认内存限制为128Mi，但在管理大量Certificate资源(>50)时，
  内存占用会显著增长(主要来自Informer缓存和ACME账户状态)
- 当集群中Certificate资源数量增加时，cert-manager需要维护更多Watch连接，
  导致内存从~80Mi飙升至200Mi+触发OOMKilled
- OOM后Pod重启，但重启期间ACME challenge可能已超时，导致续期失败
- 由于缺少告警机制，运维团队未能及时发现续期失败

**预防措施**:
- 监控cert-manager Pod的OOM事件: `kube_pod_container_status_last_terminated_reason{reason="OOMKilled"}`
- 根据Certificate资源数量合理设置内存限制(建议256Mi-512Mi)
- 部署Prometheus规则监控证书过期时间，提前14天告警

**排查过程**:
```bash
# 检查证书状态
kubectl get certificate -A
kubectl describe certificate ecommerce-tls -n production

# 检查cert-manager日志
kubectl logs -n cert-manager -l app=cert-manager --tail=100

# 检查Pod状态
kubectl get pods -n cert-manager
# 发现cert-manager-xxx OOMKilled

# 手动检查证书过期时间
echo | openssl s_client -connect www.example.com:443 2>/dev/null | openssl x509 -noout -dates
```

**解决方案**:
```bash
# 1. 立即手动续期
kubectl delete secret ecommerce-tls-secret -n production
# cert-manager会自动重新签发

# 2. 增加cert-manager内存限制
kubectl patch deployment cert-manager -n cert-manager -p '{"spec":{"template":{"spec":{"containers":[{"name":"cert-manager","resources":{"limits":{"memory":"256Mi"}}}]}}}}'

# 3. 添加证书过期告警
cat > /tmp/cert-alert.yaml << 'EOF'
- alert: TLSCertExpiringSoon
  expr: (certmanager_certificate_expiration_timestamp_seconds - time()) / 86400 < 14
  for: 1h
  labels:
    severity: warning
  annotations:
    summary: "TLS证书将在 {{ $value }} 天后过期"
- alert: TLSCertExpired
  expr: certmanager_certificate_expiration_timestamp_seconds < time()
  for: 0m
  labels:
    severity: critical
  annotations:
    summary: "TLS证书已过期!"
EOF
```

### 案例3: SSH暴力破解导致服务器被入侵

**故障现象**: 安全监控检测到某台服务器CPU使用率100%，疑似挖矿程序

**根因分析**:
- 服务器SSH端口暴露在公网
- 密码过于简单(root/123456)
- fail2ban未安装，无暴力破解防护

**排查过程**:
```bash
# 检查异常进程
ps aux | grep -E "xmrig|kdevtmpfsi|kinsing"

# 检查SSH登录日志
grep "Failed password" /var/log/secure | tail -20
grep "Accepted password" /var/log/secure | tail -20

# 检查定时任务
crontab -l
ls -la /var/spool/cron/
cat /etc/crontab

# 检查网络连接
netstat -antp | grep ESTABLISHED
ss -antp | grep -E ":4444|:3333|:8888"
```

**解决方案**:
```bash
#!/bin/bash
# emergency_response.sh - 紧急响应脚本

# 1. 立即断开网络(保留管理口)
# [警告] firewall-cmd --panic-on 会立即切断所有网络连接，包括当前SSH会话！
# 如果通过远程SSH操作，执行后将立即失去连接，可能导致锁死服务器。
# 更安全的替代方案: 仅放行管理端口(如SSH端口)，而非完全断网
firewall-cmd --panic-on  # 仅在本地控制台操作时使用

# 安全替代方案(推荐通过远程SSH使用):
firewall-cmd --permanent --remove-service=ssh && firewall-cmd --reload
# 或仅放行管理IP:
# firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="管理IP/32" port port="22" protocol="tcp" accept'
# firewall-cmd --reload

# 2. 终止恶意进程
kill -9 $(pgrep -f xmrig) 2>/dev/null
kill -9 $(pgrep -f kinsing) 2>/dev/null
kill -9 $(pgrep -f kdevtmpfsi) 2>/dev/null

# 3. 清除恶意文件
rm -rf /tmp/.xmrig /tmp/.kinsing /var/tmp/.kinsing

# 4. 清除恶意定时任务
crontab -r
rm -f /var/spool/cron/root

# 5. 重置root密码
passwd root
# 输入新密码

# 6. 禁用密码登录，仅允许密钥
sed -i 's/^PasswordAuthentication yes/PasswordAuthentication no/' /etc/ssh/sshd_config
systemctl restart sshd

# 7. 安装fail2ban
yum install -y fail2ban
cat > /etc/fail2ban/jail.local << 'EOF'
# fail2ban需配置监听SSH端口2222(与sshd_config一致)
[sshd]
enabled = true
# 使用变量而非硬编码端口号，确保与sshd_config中的Port一致
port = ${SSH_PORT:-22}  # 通过环境变量SSH_PORT指定，未设置时默认22
filter = sshd
logpath = /var/log/secure
maxretry = 3
bantime = 3600
findtime = 600
EOF
systemctl enable fail2ban
systemctl start fail2ban

# 8. 恢复网络
firewall-cmd --panic-off
```

### 案例5: 等保检查发现数据库未加密存储

**故障现象**: 等保三级测评中，测评机构指出用户密码明文存储

**根因分析**:
- 应用代码使用MD5哈希密码，未加盐
- 数据库中password字段为VARCHAR(32)，存储MD5值
- 违反等保三级"数据保密性"要求

**解决方案**:
```python
# password_hash.py - 密码加密升级方案
import hashlib
import os
import pymysql

def upgrade_passwords():
    """将MD5密码迁移到bcrypt"""
    conn = pymysql.connect(host='localhost', user='admin', password='xxx', db='users')
    cursor = conn.cursor()
    
    # 读取所有用户
    cursor.execute("SELECT id, password FROM users")
    users = cursor.fetchall()
    
    for user_id, old_md5 in users:
        # 生成新的bcrypt哈希
        import bcrypt
        # 注意: 无法从MD5反推原始密码，此处仅作为迁移框架
        # 实际方案: 1)强制用户下次登录时修改密码 2)或使用密码重置链接
        # 方案A: 标记需要重置密码(推荐)
        # new_hash = "FORCE_RESET"  # 应用层检测到此标记时强制用户修改密码
        # 方案B: 使用随机临时密码(不推荐，用户体验差)
        import secrets
        temp_password = secrets.token_urlsafe(16)
        new_hash = bcrypt.hashpw(temp_password.encode(), bcrypt.gensalt())
        
        cursor.execute(
            "UPDATE users SET password = %s, password_version = 2 WHERE id = %s",
            (new_hash.decode(), user_id)
        )
    
    conn.commit()
    print(f"✅ 迁移完成: {len(users)} 个用户")
```

```sql

-- 2. 添加加密字段
ALTER TABLE users ADD COLUMN password_encrypted VARBINARY(255);

-- 使用变量引用加密密钥，禁止硬编码密钥在SQL中
```

### 案例7: API接口未限流导致服务雪崩

**故障现象**: 活动期间API响应时间从50ms飙升到30s，最终服务不可用

**根因分析**:
- API网关未配置限流规则
- 某客户端脚本异常，每秒发送1000+请求
- 后端服务被压垮，触发级联故障

**排查过程**:
```bash
# 查看API请求量
curl -s "http://prometheus:9090/api/v1/query?query=rate(http_requests_total[5m])" | jq '.data.result[] | select(.metric.uri=="/api/v1/orders")'

# 查看Nginx连接数
ss -s | grep estab

# 查看后端服务状态
kubectl top pods -n production
```

**解决方案**:
```nginx
# nginx限流配置
http {
    # 定义限流区域
    limit_req_zone $binary_remote_addr zone=api_limit:10m rate=100r/s;
    limit_req_zone $binary_remote_addr zone=order_limit:10m rate=10r/s;
    
    server {
        # API通用限流
        location /api/ {
            limit_req zone=api_limit burst=20 nodelay;
            limit_req_status 429;
            
            proxy_pass http://backend;
        }
        
        # 订单接口严格限流
        location /api/v1/orders {
            limit_req zone=order_limit burst=5 nodelay;
            limit_req_status 429;
            
            proxy_pass http://backend;
        }
    }
}
```

```yaml
    nginx.ingress.kubernetes.io/custom-http-errors: "429"
```

### 案例8: 审计日志被恶意篡改

**故障现象**: 安全事件调查时发现审计日志不完整，部分关键记录缺失

**根因分析**:
- auditd规则未持久化，重启后丢失
- 攻击者在获取root权限后清除了/var/log/audit/
- 日志未集中存储，仅本地保存

**排查过程**:
```bash
# 检查auditd状态
systemctl status auditd
auditctl -l

# 检查日志文件完整性
sha256sum /var/log/audit/audit.log
ls -la /var/log/audit/

# 检查日志是否有被删除痕迹
ausearch -k logins --start today | tail -20
```

**解决方案**:
```bash
#!/bin/bash
# audit_fix.sh - 审计日志加固

# 1. 持久化audit规则
cat > /etc/audit/rules.d/immutable.rules << 'EOF'
# 使规则不可变(需要重启才能修改)
-e 2
EOF

# 2. 配置日志集中存储
cat >> /etc/rsyslog.conf << 'EOF'
# 审计日志发送到远程SIEM
local6.* @@siem.internal.com:514
EOF

# 3. 安装日志完整性监控
yum install -y aide
aide --init
mv /var/lib/aide/aide.db.new.gz /var/lib/aide/aide.db.gz

# 4. 配置日志轮转
cat > /etc/logrotate.d/audit << 'EOF'
/var/log/audit/*.log {
    daily
    rotate 90
    compress
    delaycompress
    notifempty
    create 0600 root root
    postrotate
        /sbin/service auditd restart 2>/dev/null || true
    endscript
}
EOF

# 5. 启用auditd
systemctl enable auditd
systemctl start auditd
```

```yaml
# compliance-monitoring.yml
groups:
  - name: compliance-monitoring
    rules:
      # 密码策略检查
      - alert: PasswordPolicyViolation
        expr: node_password_policy_max_days < 90
        for: 0m
        labels:
          severity: warning
          type: compliance
        annotations:
          summary: "密码策略不符合等保要求"
          
      # SSH配置检查
      - alert: SSHConfigViolation
        expr: node_ssh PermitRootLogin != "no" or node_ssh PasswordAuthentication != "no"
        for: 0m
        labels:
          severity: critical
          type: compliance
        annotations:
          summary: "SSH配置不符合等保要求"
          
      # 审计日志检查
      - alert: AuditLogMissing
        expr: node_auditd_status != "active"
        for: 5m
        labels:
          severity: critical
          type: compliance
        annotations:
          summary: "审计服务未运行"
          
      # 防火墙状态检查
      - alert: FirewallDisabled
        expr: node_firewalld_status != "active"
        for: 1m
        labels:
          severity: critical
          type: compliance
        annotations:
          summary: "防火墙未启用"
          
      # 证书过期检查
      - alert: CertificateExpiringSoon
        expr: (certmanager_certificate_expiration_timestamp_seconds - time()) / 86400 < 30
        for: 1h
        labels:
          severity: warning
          type: compliance
        annotations:
          summary: "TLS证书将在 {{ $value }} 天后过期"
```

### 13.4 等保合规自动化检查脚本

```bash
#!/bin/bash
# compliance_check.sh - 等保合规自动化检查

set -euo pipefail

REPORT_FILE="/tmp/compliance_report_$(date +%Y%m%d).txt"
echo "等保三级合规检查报告" > $REPORT_FILE
echo "检查时间: $(date)" >> $REPORT_FILE
echo "========================================" >> $REPORT_FILE

# 1. SSH配置检查
echo "1. SSH配置检查" >> $REPORT_FILE
if grep -q "^PermitRootLogin no" /etc/ssh/sshd_config && \
   grep -q "^PasswordAuthentication no" /etc/ssh/sshd_config && \
   grep -q "^MaxAuthTries 3" /etc/ssh/sshd_config; then
    echo "   ✅ SSH配置符合要求" >> $REPORT_FILE
else
    echo "   ❌ SSH配置不符合要求" >> $REPORT_FILE
fi

# 2. 防火墙状态检查
echo "2. 防火墙状态检查" >> $REPORT_FILE
if systemctl is-active firewalld 2>/dev/null | grep -q "active"; then  # K8s节点应关闭
    echo "   ✅ 防火墙已启用" >> $REPORT_FILE
else
    echo "   ⚠️ 防火墙未启用(K8s节点建议关闭)" >> $REPORT_FILE
fi

# 3. 审计服务检查
echo "3. 审计服务检查" >> $REPORT_FILE
if systemctl is-active auditd | grep -q "active"; then
    echo "   ✅ 审计服务已启用" >> $REPORT_FILE
else
    echo "   ❌ 审计服务未启用" >> $REPORT_FILE
fi

# 4. 文件权限检查
echo "4. 文件权限检查" >> $REPORT_FILE
PERM_ISSUES=0
for file in /etc/shadow /etc/passwd /etc/ssh/sshd_config; do
    if [ -f "$file" ]; then
        PERM=$(stat -c %a $file)
        echo "   $file: $PERM" >> $REPORT_FILE
    fi
done

# 5. 用户账户检查
echo "5. 用户账户检查" >> $REPORT_FILE
EMPTY_PASS=$(awk -F: '($2 == "" || $2 == "!") {print $1}' /etc/shadow 2>/dev/null | wc -l)
echo "   无密码账户数: $EMPTY_PASS" >> $REPORT_FILE
if [ "$EMPTY_PASS" -gt 0 ]; then
    echo "   ⚠️ 存在无密码账户" >> $REPORT_FILE
fi

# 6. 特权账户检查
echo "6. 特权账户检查" >> $REPORT_FILE
SUID_COUNT=$(find / -type f -perm -4000 2>/dev/null | wc -l)
echo "   SUID文件数: $SUID_COUNT" >> $REPORT_FILE

# 7. 定时任务检查
echo "7. 定时任务检查" >> $REPORT_FILE
CRON_COUNT=$(crontab -l 2>/dev/null | grep -v "^#" | wc -l)
echo "   定时任务数: $CRON_COUNT" >> $REPORT_FILE

echo "========================================" >> $REPORT_FILE
echo "检查完成: $(date)" >> $REPORT_FILE

cat $REPORT_FILE
```
