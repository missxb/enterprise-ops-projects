    38|#!/bin/bash
    39|# hardening.sh - CentOS 7/8 主机安全加固
    40|
    41|set -euo pipefail
    42|
    43|echo "========== 1. 账户安全 =========="
    44|# 禁用root远程登录（保留sudo）
    45|sed -i 's/^#PermitRootLogin yes/PermitRootLogin prohibit-password/' /etc/ssh/sshd_config
    46|sed -i 's/^PasswordAuthentication yes/PasswordAuthentication no/' /etc/ssh/sshd_config
    47|sed -i 's/^#MaxAuthTries 6/MaxAuthTries 3/' /etc/ssh/sshd_config
    48|sed -i 's/^#ClientAliveInterval 0/ClientAliveInterval 300/' /etc/ssh/sshd_config
    49|sed -i 's/^#ClientAliveCountMax 3/ClientAliveCountMax 2/' /etc/ssh/sshd_config
    50|systemctl restart sshd
    51|
    52|# 密码策略
    53|cat >> /etc/login.defs << 'EOF'
    54|PASS_MAX_DAYS   90
    55|PASS_MIN_DAYS   7
    56|PASS_MIN_LEN    12
    57|PASS_WARN_AGE   14
    58|EOF
    59|
    60|# 锁定无用账户
    61|for user in games ftp; do
    62|  passwd -l $user 2>/dev/null || true
    63|done
    64|
    65|echo "========== 2. 文件权限加固 =========="
    66|chmod 600 /etc/shadow
    67|chmod 644 /etc/passwd
    68|chmod 600 /etc/ssh/sshd_config
    69|chmod 700 /root
    70|chmod 600 /boot/grub2/grub.cfg
    71|
    72|# 查找并修复SUID/SGID文件
    73|find / -type f \( -perm -4000 -o -perm -2000 \) -exec ls -la {} \; 2>/dev/null
    74|
    75|echo "========== 3. 内核安全加固 =========="
    76|cat >> /etc/sysctl.d/99-security.conf << 'EOF'
    77|# 禁止IP转发（非路由器）
    78|net.ipv4.ip_forward = 0
    79|
    80|# 禁止ICMP重定向
    81|net.ipv4.conf.all.accept_redirects = 0
    82|net.ipv4.conf.all.send_redirects = 0
    83|
    84|# 禁止源路由
    85|net.ipv4.conf.all.accept_source_route = 0
    86|
    87|# SYN Flood防护
    88|net.ipv4.tcp_syncookies = 1
    89|net.ipv4.tcp_max_syn_backlog = 65535
    90|net.ipv4.tcp_synack_retries = 2
    91|
    92|# 禁止ICMP广播
    93|net.ipv4.icmp_echo_ignore_broadcasts = 1
    94|
    95|# 记录可疑数据包
    96|net.ipv4.conf.all.log_martians = 1
    97|
    98|# 反向路径过滤
    99|net.ipv4.conf.all.rp_filter = 1
   100|
   101|# 禁止IP伪造
   102|net.ipv4.conf.all.accept_redirects = 0
   103|net.ipv6.conf.all.accept_redirects = 0
   104|EOF
   105|sysctl --system
   106|
   107|echo "========== 4. 防火墙配置 =========="
   108|# 使用firewalld而非iptables
   109|systemctl enable firewalld
   110|systemctl start firewalld
   111|
   112|# 默认拒绝所有
   113|firewall-cmd --set-default-zone=drop
   114|
   115|# 允许SSH
   116|firewall-cmd --permanent --add-service=ssh
   117|
   118|# 允许HTTP/HTTPS
   119|firewall-cmd --permanent --add-port=80/tcp
   120|firewall-cmd --permanent --add-port=443/tcp
   121|
   122|# 允许内网管理
   123|firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="10.10.0.0/16" accept'
   124|

   126|firewall-cmd --reload
   127|
   128|echo "========== 5. 审计日志 =========="
   129|yum install -y audit
   130|systemctl enable auditd
   131|systemctl start auditd
   132|
   133|# 审计规则
   134|cat > /etc/audit/rules.d/hardening.rules << 'EOF'
   135|-w /etc/passwd -p wa -k identity
   136|-w /etc/shadow -p wa -k identity
   137|-w /etc/group -p wa -k identity
   138|-w /etc/sudoers -p wa -k sudoers
   139|-w /var/log/secure -p wa -k logins
   140|-w /var/log/wtmp -p wa -k logins
   141|-w /var/log/btmp -p wa -k logins
   142|-w /etc/ssh/sshd_config -p wa -k sshd_config
   143|-a always,exit -F arch=b64 -S execve -C uid!=euid -F euid=0 -k priv_escalation
   144|-a always,exit -F arch=b64 -S connect,accept,bind -F exit=-EACCES -k network
   145|-a always,exit -F arch=b64 -S open,creat,truncate,ftruncate -F exit=-EACCES -k access

```
安全事件响应流程(SOP):

1. 检测阶段 (0-15分钟)
   ├── 确认事件真实性
   ├── 评估影响范围
   ├── 通知安全团队
   └── 记录事件时间线

2. 遏制阶段 (15-60分钟)
   ├── 隔离受影响系统
   ├── 封禁攻击IP
   ├── 保留证据(日志/快照)
   └── 阻止横向移动

3. 消除阶段 (1-24小时)
   ├── 修复漏洞
   ├── 清除恶意软件
   ├── 重置泄露凭证
   └── 更新安全策略

4. 恢复阶段 (24-72小时)
   ├── 恢复系统服务
   ├── 验证数据完整性
   ├── 监控异常行为
   └── 逐步恢复业务

5. 总结阶段 (72小时后)
   ├── 编写事件报告
   ├── 复盘改进措施
   ├── 更新应急预案
   └── 安全培训
```
